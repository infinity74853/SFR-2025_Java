3    public class DepositCalculator {
4      public static double calculateDeposit(double initialAmount, double monthlyRate, int years) {
5        double amount = initialAmount;
6        int months = years * 12;
7        for (int i = 0; i < months; i++) {
8          amount += amount * (monthlyRate / 100);
9        }
10       return amount;
11     }
