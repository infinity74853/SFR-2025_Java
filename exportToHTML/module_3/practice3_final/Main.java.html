109    private static void initFlights(TicketBookingSystem system) {
110      Crew crew = new Crew("Иванов", "Петров", new String[]{"Сидорова", "Кузнецов"});
111      LocalDateTime depTime = LocalDateTime.of(2024, 12, 25, 15, 30);
112      system.addFlight(new ConcreteFlight("SU123", "Boeing 737", depTime,
113          "Москва", "Санкт-Петербург", crew, 5000, 15000));
114    }

116    private static List<Flight> searchFlights(TicketBookingSystem system, Scanner scanner) {
117      System.out.print("Откуда: ");
118      String from = scanner.next();
119      System.out.print("Куда: ");
120      String to = scanner.next();
121      System.out.print("Дата (yyyy-mm-dd): ");
122      LocalDate date = LocalDate.parse(scanner.next());
123  
124      return system.searchFlights(from, to, date);
125    }

127    private static void buyTicket(TicketBookingSystem system, Scanner scanner) {
128      System.out.print("Дата рождения (yyyy-mm-dd): ");
129      LocalDate birthDate = LocalDate.parse(scanner.next());
130      Passenger passenger = new Passenger(birthDate);
131  
132      List<Flight> flights = searchFlights(system, scanner);
133      if (flights.isEmpty()) {
134        System.out.println("Рейсы не найдены!");
135        return;
136      }
137  
138      Flight flight = flights.get(0);
139      System.out.print("Класс (ECONOMY/BUSINESS): ");
140      String ticketClass = scanner.next().toUpperCase();

142      Ticket ticket = system.sellTicket(passenger, flight, ticketClass, LocalDate.now());
