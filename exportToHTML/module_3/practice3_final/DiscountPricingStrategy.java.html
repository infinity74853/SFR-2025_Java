3    import java.time.LocalDate;
4    import java.time.temporal.ChronoUnit;
5    
6    public class DiscountPricingStrategy implements TicketPricingStrategy {
7      @Override
8      public double calculatePrice(double basePrice, LocalDate purchaseDate, LocalDate flightDate) {
9        long daysBefore = ChronoUnit.DAYS.between(purchaseDate, flightDate);
10       if (daysBefore > 30) return basePrice * 0.8;    // 20% скидка
11       if (daysBefore < 7) return basePrice * 1.2;      // 20% наценка
12       return basePrice;
13     }
14   }
