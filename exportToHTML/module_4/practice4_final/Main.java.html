23   import java.io.File;
24   import java.io.IOException;
25   import java.util.Scanner;

27   public class Main {
28     public static void main(String[] args) {
29       Scanner scanner = new Scanner(System.in);
30       BookStore store;
31   
32       try {
33         store = BookStore.loadState("bookstore.dat");
34         System.out.println("Данные успешно загружены");
35       } catch (ClassNotFoundException | IOException e) {
36         System.out.println("Ошибка загрузки: " + e.getMessage());
37         store = new BookStore();
38       }
39   
40       while (true) {
41         System.out.println("\n1. Добавить книгу\n2. Оформить заказ\n3. Выход");
42         int choice = scanner.nextInt();
43         scanner.nextLine();
44   
45         switch (choice) {
46           case 1 -> {
47             System.out.print("Автор: ");
48             String author = scanner.nextLine();
49             System.out.print("Название: ");
50             String title = scanner.nextLine();
51             System.out.print("Издательство: ");
52             String publisher = scanner.nextLine();
53             System.out.print("Год: ");
54             int year = scanner.nextInt();
55             System.out.print("Цена: ");
56             double price = scanner.nextDouble();
57             scanner.nextLine();
58   
59             store.addBook(new Book(author, title, publisher, year, price));
60           }
61           case 2 -> {
62             System.out.print("Ваше имя: ");
63             String name = scanner.nextLine();
64             System.out.print("Email: ");
65             String email = scanner.nextLine();
66             System.out.print("Название книги: ");
67             String bookTitle = scanner.nextLine();
68   
69             Customer customer = new Customer(name, email);
70             store.processOrder(bookTitle, customer);
71           }
72           case 3 -> {
73             try {
74               store.saveState("bookstore.dat");
75               System.out.println("Данные сохранены в: "
76                   + new File("bookstore.dat").getAbsolutePath());
77               return;
78             } catch (IOException e) {
79               System.err.println("Ошибка сохранения: " + e.getMessage());
80             }
81           }
82           default -> System.out.println("Неверный выбор!");
83         }
84       }
85     }
86   }
